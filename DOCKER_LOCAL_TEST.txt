# Edit project root .env: set MYSQL_PASSWORD, MYSQL_ROOT_PASSWORD (optional: REDIS_URL for external Redis).
# LLM / ADMIN_API_KEY / JWT / etc. live in factory-agent\.env and emas\.env — Compose merges them

docker compose build
docker compose up -d

# Seed MySQL (canonical demo data). Uses same DB_* env as go-api (host mysql in Compose).
docker compose run --rm --entrypoint /usr/local/bin/seed go-api

docker compose ps
docker compose logs -f --tail=100 nginx go-api factory-agent

# Health checks
curl http://localhost/health
curl http://localhost/api/v1/machines
curl http://localhost/agent/health

# Stop
docker compose down

# Stop and delete MySQL volume
docker compose down -v
